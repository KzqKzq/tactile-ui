# Alert 警告

警告用于向用户展示重要的提示信息。

## 基础用法

```tsx preview
import { Alert } from '@kzqkzq/tactile-ui';

export default function AlertDemo() {
  return (
    <div style={{ display: 'grid', gap: 12, width: '100%' }}>
      <Alert variant="info" title="提示" description="这是一条信息提示" />
      <Alert variant="success" title="成功" description="操作已完成" />
      <Alert variant="warning" title="警告" description="请注意这个问题" />
      <Alert variant="error" title="错误" description="操作失败，请重试" />
    </div>
  );
}
```

## 可关闭

```tsx preview
import { useState } from 'react';
import { Alert } from '@kzqkzq/tactile-ui';

export default function ClosableAlert() {
  const [open, setOpen] = useState(true);
  return open ? (
    <Alert
      variant="info"
      title="可关闭的提示"
      description="点击右侧按钮关闭"
      closable
      onClose={() => setOpen(false)}
    />
  ) : (
    <div>提示已关闭</div>
  );
}
```

## 实心样式

```tsx preview
import { Alert } from '@kzqkzq/tactile-ui';

export default function SolidAlertDemo() {
  return (
    <div style={{ display: 'grid', gap: 12, width: '100%' }}>
      <Alert variant="info" solid title="Info" />
      <Alert variant="success" solid title="Success" />
      <Alert variant="warning" solid title="Warning" />
      <Alert variant="error" solid title="Error" />
    </div>
  );
}
```

## API

### Props

| 属性 | 说明 | 类型 | 默认值 |
|------|------|------|--------|
| variant | 类型 | `'info' \| 'success' \| 'warning' \| 'error'` | `'info'` |
| title | 标题 | `string` | - |
| description | 描述 | `string` | - |
| closable | 是否可关闭 | `boolean` | `false` |
| solid | 是否实心 | `boolean` | `false` |
| icon | 自定义图标 | `ReactNode` | - |
| onClose | 关闭回调 | `() => void` | - |
