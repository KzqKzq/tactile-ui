# Progress 进度条

进度条用于展示操作进度。

## 基础用法

```tsx preview
import { Progress } from '@kzqkzq/tactile-ui';

export default function ProgressBasic() {
  return (
    <div style={{ width: 320 }}>
      <Progress value={65} />
    </div>
  );
}
```

## 带标签

```tsx preview
import { Progress } from '@kzqkzq/tactile-ui';

export default function ProgressWithLabel() {
  return (
    <div style={{ width: 320 }}>
      <Progress value={45} label="下载进度" showValue />
    </div>
  );
}
```

## 圆形进度

```tsx preview
import { CircularProgress } from '@kzqkzq/tactile-ui';

export default function CircularProgressDemo() {
  return (
    <div style={{ display: 'flex', gap: 12, alignItems: 'center' }}>
      <CircularProgress value={75} size="sm" />
      <CircularProgress value={75} />
      <CircularProgress value={75} size="lg" showValue />
    </div>
  );
}
```

## API

### Progress Props

| 属性 | 说明 | 类型 | 默认值 |
|------|------|------|--------|
| value | 进度值 | `number` | `0` |
| max | 最大值 | `number` | `100` |
| size | 尺寸 | `'sm' \| 'md' \| 'lg'` | `'md'` |
| label | 标签 | `string` | - |
| showValue | 显示数值 | `boolean` | `false` |
| formatValue | 格式化函数 | `(value: number, max: number) => string` | - |
